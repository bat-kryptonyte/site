OurComparable Spider

public class Spider implements OurComparable {
	private int legs;
	public Spider(int x) {
		if (x <= 0) {
			throw new IllegalArgumentException();
		}
		legs = x;
	}
	public int compareTo(Object a) {
		if (!(a instanceof Spider)) {
			throw new IllegalArgumentException();
		} else if (legs < ((Spider) a).legs) {
			return -1;
		} else if (legs > ((Spider) a).legs) {
			return 1;
		}
		return 0;
	}
}

OurComparable Count Less Than

public class CountLessThan {
	public static int count(OurComparable[] x, OurComparable y) {
		assert x != null;
		assert y != null;
		int count = 0;
		for (OurComparable i: x) {
			if (i.compareTo(y) < 0) {
				count++;
			}
		}
		return count;
	}
}

Different Count

public class DifferentCount {
	public static int count(Object[] a, Object b) {
		assert a != null;
		int count = 0;
		for (Object i: a) {
			if(!(i == b)) {
				count++;
			}
		}
		return count;
	}
}
