Longest String

public class Longest {
	private int count;
	private String x;
	public Longest(String y) {
		assert y != null;
		count = y.length();
		x = y;
	}
	public void add(String y) {
		try {
			if (y.length() > count) {
				count = y.length();
				x = y;
			}
		} catch(NullPointerException e) {
			 throw e;
		}
	public String current() {
		return x;
	}
}

Dog Is Younger

public class Dog {
	private int age;
	public Dog(int x) {
		assert x >= 0;
		age = x;
	}
	public static boolean isYounger(Dog a, Dog b) {
		assert a != null;
		assert b != null;
		return a.age < b.age;
	}
}
