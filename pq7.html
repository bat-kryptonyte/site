Polymorphism and References

Cat Equality

public class Cat {
	private String name;
	private String breed;
	private int lives;
	public Cat(String x, String y, int z) {
		assert x != null;
		assert y != null;
		assert z > 0 && z <= 9;
		name = x;
		breed = y;
		lives = z;
	}
	@Override
	public boolean equals(Object a) {
		if (!(a instanceof Cat)) {
			return false;
		}
		return name.equals(((Cat)a).name) && breed.equals(((Cat)a).breed);
		// ^non primitive types so use .equals() to compare
	}
}

Flammable Starter

public class Starter {
	public static String start(Flammable x) {
		if (x instanceof Match) {
			return ((Match) x).strike();
		} else if (x instanceof Leaves) {
			return ((Leaves) x).ember();
		} else if (x instanceof Grill) {
			return ((Grill) x).charcoal();
		}
		return null;
	}
}

All or Nothing

public class All {
	public static Item[] orNothing(int count, boolean x) {
		Item[] a = new Item[count];
		Item b = new Item();
		if (x) {
			for (int i = 0; i < count; i++) {
				a[i] = (Item) b;
			}
		} else {
			for (int i = 0; i < count; i++) {
				a[i] = new Item();
			}
		}
		return a;
	}
}
